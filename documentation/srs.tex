\subsection{RADAR Sensing}

\requirement
  {FR-001}
  {RADAR Detection Range}
  {The RADAR \textbf{shall} have a maximum detection range of $10\ \text{m}$.}
  {Defines the operational sensing envelope required for target detection within the intended deployment environment.}
  {High}
  {Sean Carson}

\requirement
  {FR-002}
  {RADAR Range Resolution}
  {The RADAR \textbf{shall} achieve a range resolution of $0.5\ \text{m}$.}
  {Sufficient range resolution is required to distinguish between closely spaced targets along the radial axis.}
  {High}
  {Sean Carson}

\requirement
  {FR-003}
  {RADAR Angular Resolution}
  {The RADAR \textbf{shall} achieve an angular resolution of $0.02499\ \text{rad}$ ($\approx 1.43^{\circ}$).}
  {Angular resolution determines the ability to discriminate between targets at similar ranges but different bearings.}
  {High}
  {Sean Carson}

\requirement
  {FR-004}
  {RADAR Field of View}
  {The RADAR \textbf{shall} provide a field of view of $\frac{2\pi}{3}\ \text{rad}$ ($120^{\circ}$).}
  {A $120^{\circ}$ FOV is required to cover the designated surveillance sector without mechanical scanning.}
  {High}
  {Sean Carson}

\subsection{Data Output and Display}

\requirement
  {FR-005}
  {Live GUI Display}
  {The RADAR \textbf{shall} output all collected data to a live graphical user interface (GUI) display in real time.}
  {Operators require a continuous, real-time visual representation of RADAR returns to monitor the environment and respond to detections promptly.}
  {High}
  {Sean Carson}
